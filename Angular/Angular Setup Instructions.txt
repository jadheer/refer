
npm install @angular/cli -g

ng new [project-name]

// Version
ng --version

// To test the possible
ng new example --dry-run

// To add a component
ng generate component [component_name]
ng g c [component_name]

// To add a class
ng g class [class_name]

// To add a service
ng g service [service_name]

// Run
ng serve
